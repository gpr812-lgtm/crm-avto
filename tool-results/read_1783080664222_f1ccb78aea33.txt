   891→            statusEl.className = 'backup-status warning';
   892→            textEl.textContent = `⚠️ Бэкап устарел (${Math.floor(hoursDiff)}ч)`;
   893→        } else {

   895→            textEl.textContent = `💾 ${backupDate.toLocaleString('ru-RU', {day:'2-digit', month:'2-digit', hour:'2-digit', minute:'2-digit'})}`;
   896→        }

   962→function importFile(event, tabKey) {
   963→    const file = event.target.files[0];
   964→    if (!file) return;
   965→    const ext = file.name.split('.').pop().toLowerCase();
   966→    if (ext === 'csv') {
   967→        importCSV(event, tabKey);
   968→    } else if (ext === 'xlsx' || ext === 'xls') {
   969→        const reader = new FileReader();
   970→        reader.onload = e => {
   971→            try {
   972→                const data = new Uint8Array(e.target.result);
   973→                const wb = XLSX.read(data, { type: 'array' });
   974→                const ws = wb.Sheets[wb.SheetNames[0]];
   975→                const json = XLSX.utils.sheet_to_json(ws, { header: 1 });
   976→                if (json.length < 2) { showToast('Файл пуст', 'error'); return; }
   977→                const imported = [];
   978→                for (let i = 1; i < json.length; i++) {
   979→                    const rowData = json[i];
   980→                    if (!rowData || rowData.length === 0) continue;
   981→                    const row = { id: Date.now() + i };
   982→                    tabsState[tabKey].columns.forEach((c, idx) => {
   983→                        const val = rowData[idx + 1];
   984→                        if (val !== undefined) {
   985→                            if (c.type === 'number') row[c.key] = parseFloat(val) || 0;
   986→                            else row[c.key] = String(val);
   987→                        } else row[c.key] = c.default;
   988→                    });
   989→                    imported.push(row);
   990→                }
   991→                if (imported.length > 0 && confirm(`Импортировать ${imported.length} строк?`)) {
   992→                    tabsState[tabKey].data = imported;
   993→                    saveTabDataKey(tabKey);
   994→                    renderTab(tabKey);
   995→                    showToast(`📤 Импортировано ${imported.length} строк`);
   996→                    addHistoryEntry('bulk', `Импорт ${imported.length} строк из Excel`);
   997→                }
   998→            } catch (err) { showToast('❌ Ошибка: ' + err.message, 'error'); }
   999→        };
  1000→        reader.readAsArrayBuffer(file);
  1001→    } else {
  1002→        showToast('Неподдерживаемый формат', 'error');
  1003→    }
  1004→    event.target.value = '';
  1005→}

  1183→function calculateForecast() {
  1184→    const monthValue = document.getElementById('traffic-month')?.value || new Date().toISOString().slice(0, 7);
  1185→    const [year, month] = monthValue.split('-').map(Number);
  1186→    const daysInMonth = getDaysInMonth(month - 1, year);
  1187→    const today = new Date();
  1188→    const isCurrentMonth = today.getFullYear() === year && today.getMonth() === month - 1;
  1189→    const daysPassed = isCurrentMonth ? today.getDate() : daysInMonth;
  1190→    const skladData = tabsState.sklad?.data || [];
  1191→    let contractsSoFar = 0;
  1192→    skladData.forEach(r => {
  1193→        if (r.status !== 'Продан' || !r.dateDkp) return;
  1194→        const d = new Date(r.dateDkp);
  1195→        if (d.getFullYear() === year && d.getMonth() === month - 1 && d.getDate() <= daysPassed) contractsSoFar++;
  1196→    });
  1197→    const dailyAverage = daysPassed > 0 ? contractsSoFar / daysPassed : 0;
  1198→    const forecastTotal = Math.round(dailyAverage * daysInMonth);
  1199→    let planTotal = 0;
  1200→    for (let day = 1; day <= daysInMonth; day++) {
  1201→        const dayStr = String(day).padStart(2, '0');
  1202→        planTotal += todayPlans[monthValue]?.[dayStr]?.contracts || 0;
  1203→    }
  1204→    const prevMonth = month === 1 ? 12 : month - 1;
  1205→    const prevYear = month === 1 ? year - 1 : year;
  1206→    let prevMonthContracts = 0;
  1207→    skladData.forEach(r => {
  1208→        if (r.status !== 'Продан' || !r.dateDkp) return;
  1209→        const d = new Date(r.dateDkp);
  1210→        if (d.getFullYear() === prevYear && d.getMonth() === prevMonth - 1) prevMonthContracts++;
  1211→    });
  1212→    const trendPct = prevMonthContracts > 0 ? ((contractsSoFar / prevMonthContracts - 1) * 100).toFixed(1) : 0;
  1213→    return { contractsSoFar, daysPassed, daysInMonth, dailyAverage: dailyAverage.toFixed(1), forecastTotal, planTotal, trendPct, isCurrentMonth };
  1214→}

  1465→function getWeeksOfMonth(year, month) {
  1466→    const daysInMonth = getDaysInMonth(month - 1, year);
  1467→    const weeks = [];
  1468→    let currentWeek = { days: [] };
  1469→    for (let day = 1; day <= daysInMonth; day++) {
  1470→        const dayOfWeek = new Date(year, month - 1, day).getDay();
  1471→        const isMonday = dayOfWeek === 1 || day === 1;
  1472→        if (isMonday && currentWeek.days.length > 0) { weeks.push(currentWeek); currentWeek = { days: [] }; }
  1473→        currentWeek.days.push(day);
  1474→    }
  1475→    if (currentWeek.days.length > 0) weeks.push(currentWeek);
  1476→    return weeks;
  1477→}

  1485→function getSkladStats(monthValue) {
  1486→    const skladData = tabsState.sklad?.data || [];
  1487→    const calls = { byModel: {}, total: 0 };
  1488→    const callsStrike = { byModel: {}, total: 0 };
  1489→    const visits = { byModel: {}, total: 0 };
  1490→    const visitsStrike = { byModel: {}, total: 0 };
  1491→    skladData.forEach(r => {
  1492→        if (!r.dateDkp || !r.dateDkp.startsWith(monthValue)) return;
  1493→        const model = r.model || 'Не указано';
  1494→        const traffic = r.traffic || '';
  1495→        const status = r.status;
  1496→        const isCallOrApp = traffic.includes('Звонок') || traffic.includes('Заявка');
  1497→        const isVisit = traffic.includes('Визит');
  1498→        if (status === 'Продан') {
  1499→            if (isCallOrApp) { if (!calls.byModel[model]) calls.byModel[model] = 0; calls.byModel[model]++; calls.total++; }
  1500→            else if (isVisit) { if (!visits.byModel[model]) visits.byModel[model] = 0; visits.byModel[model]++; visits.total++; }
  1501→        } else if (status === 'Отказ') {
  1502→            if (isCallOrApp) { if (!callsStrike.byModel[model]) callsStrike.byModel[model] = 0; callsStrike.byModel[model]++; callsStrike.total++; }
  1503→            else if (isVisit) { if (!visitsStrike.byModel[model]) visitsStrike.byModel[model] = 0; visitsStrike.byModel[model]++; visitsStrike.total++; }
  1504→        }
  1505→    });
  1506→    return { calls, callsStrike, visits, visitsStrike };
  1507→}
