function renderHistory() {
    const list = document.getElementById('historyList');
    document.getElementById('historyCount').textContent = changeHistory.length;
    if (changeHistory.length === 0) {
        list.innerHTML = '<p style="color:#7f8c8d; text-align:center; padding:20px;">История пуста</p>';
        return;
    }
    const typeLabels = { add: 'Добавление', edit: 'Изменение', delete: 'Удаление', bulk: 'Массовое' };
    list.innerHTML = changeHistory.map(entry => {
        const time = new Date(entry.time).toLocaleString('ru-RU', { day:'2-digit', month:'2-digit', year:'numeric', hour:'2-digit', minute:'2-digit' });
        return `<div class="history-item"><div class="history-time">${time}</div><div class="history-action"><span class="history-type ${entry.type}">${typeLabels[entry.type] || entry.type}</span>${entry.description}</div></div>`;
    }).join('');
}

function exportHistory() {
    if (changeHistory.length === 0) { showToast('История пуста', 'warning'); return; }
    let csv = 'Дата;Тип;Описание\n';
    changeHistory.forEach(e => {
        csv += `"${new Date(e.time).toLocaleString('ru-RU')}";"${e.type}";"${e.description.replace(/"/g, '""')}"\n`;
    });
    const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `history_${new Date().toISOString().split('T')[0]}.csv`;
    a.click();
    URL.revokeObjectURL(url);
    showToast('📥 История экспортирована');
}

function updateBackupStatus() {
    const lastBackup = localStorage.getItem('crm_last_backup_time');
    const statusEl = document.getElementById('backupStatus');
    const textEl = document.getElementById('backupStatusText');
    if (!lastBackup) {
        statusEl.className = 'backup-status danger';
        textEl.textContent = '⚠️ Бэкап не создан!';
    } else {
        const backupDate = new Date(lastBackup);
        const hoursDiff = (Date.now() - backupDate) / (1000 * 60 * 60);
        if (hoursDiff > 24) {
            statusEl.className = 'backup-status warning';
            textEl.textContent = `⚠️ Бэкап устарел (${Math.floor(hoursDiff)}ч)`;
        } else {
            statusEl.className = 'backup-status';
            textEl.textContent = `💾 ${backupDate.toLocaleString('ru-RU', {day:'2-digit', month:'2-digit', hour:'2-digit', minute:'2-digit'})}`;
        }
    }
}

function exportTrafficToExcel() {
    const monthValue = document.getElementById('traffic-month').value;
    const [year, month] = monthValue.split('-').map(Number);
    const daysInMonth = getDaysInMonth(month - 1, year);
    const models = selectOptions.model || [];
    const monthData = trafficData[monthValue] || {};
    const stats = getSkladStats(monthValue);
    const wb = XLSX.utils.book_new();
    const callsHeaders = ['Модель', ...Array.from({length: daysInMonth}, (_, i) => `День ${i+1}`), 'Итого', 'Контракты', '%'];
    const callsData = [callsHeaders];
    models.forEach(model => {
        const md = monthData[model] || { callsAndApps: {}, visits: {} };
        const row = [model];
        let total = 0;
        for (let day = 1; day <= daysInMonth; day++) {
            const v = md.callsAndApps[String(day).padStart(2, '0')] || 0;
            row.push(v);
            total += v;
        }
        row.push(total);
        row.push(stats.calls.byModel[model] || 0);
        row.push(total > 0 ? ((stats.calls.byModel[model] || 0) / total * 100).toFixed(1) + '%' : '—');
        callsData.push(row);
    });
    const wsCalls = XLSX.utils.aoa_to_sheet(callsData);
    XLSX.utils.book_append_sheet(wb, wsCalls, 'Звонки');
    XLSX.writeFile(wb, `traffic_${monthValue}.xlsx`);
    showToast('📊 Excel экспортирован');
}

function importFile(event, tabKey) {
    const file = event.target.files[0];
    if (!file) return;
    const ext = file.name.split('.').pop().toLowerCase();
    if (ext === 'csv') {
        importCSV(event, tabKey);
    } else if (ext === 'xlsx' || ext === 'xls') {
        const reader = new FileReader();
        reader.onload = e => {
            try {
                const data = new Uint8Array(e.target.result);
                const wb = XLSX.read(data, { type: 'array' });
                const ws = wb.Sheets[wb.SheetNames[0]];
                const json = XLSX.utils.sheet_to_json(ws, { header: 1 });
                if (json.length < 2) { showToast('Файл пуст', 'error'); return; }
                const imported = [];
                for (let i = 1; i < json.length; i++) {
                    const rowData = json[i];
                    if (!rowData || rowData.length === 0) continue;
                    const row = { id: Date.now() + i };
                    tabsState[tabKey].columns.forEach((c, idx) => {
                        const val = rowData[idx + 1];
                        if (val !== undefined) {
                            if (c.type === 'number') row[c.key] = parseFloat(val) || 0;
                            else row[c.key] = String(val);
                        } else row[c.key] = c.default;
                    });
                    imported.push(row);
                }
                if (imported.length > 0 && confirm(`Импортировать ${imported.length} строк?`)) {
                    tabsState[tabKey].data = imported;
                    saveTabDataKey(tabKey);
                    renderTab(tabKey);
                    showToast(`📤 Импортировано ${imported.length} строк`);
                    addHistoryEntry('bulk', `Импорт ${imported.length} строк из Excel`);
                }
            } catch (err) { showToast('❌ Ошибка: ' + err.message, 'error'); }
        };
        reader.readAsArrayBuffer(file);
    } else {
        showToast('Неподдерживаемый формат', 'error');
    }
    event.target.value = '';
}

function performGlobalSearch() {
    const query = document.getElementById('globalSearchInput').value.toLowerCase().trim();
    const results = document.getElementById('globalSearchResults');
    if (!query) { results.innerHTML = ''; results.classList.remove('active'); return; }
    const found = [];
    tabsState.sklad.data.forEach(row => {
        const text = Object.values(row).join(' ').toLowerCase();
        if (text.includes(query)) {
            found.push({ tab: 'Склад', tabKey: 'sklad', rowId: row.id, match: highlightMatch(`${row.model || ''} — ${row.client || ''}`, query) });
        }
    });
    if (found.length === 0) {
        results.innerHTML = '<div class="search-result-item" style="color:#7f8c8d;">Ничего не найдено</div>';
    } else {
        results.innerHTML = found.slice(0, 20).map(item => `<div class="search-result-item" onclick="goToSearchResult('${item.tabKey}', ${item.rowId})"><strong style="color:#2a5298;">[${item.tab}]</strong> ${item.match}</div>`).join('');
    }
    results.classList.add('active');
}

function highlightMatch(text, query) {
    const regex = new RegExp(`(${query.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')})`, 'gi');
    return text.replace(regex, '<span class="highlight">$1</span>');
}

function goToSearchResult(tabKey, rowId) {
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
    document.querySelector(`[data-tab="${tabKey}"]`).classList.add('active');
    document.getElementById(`tab-${tabKey}`).classList.add('active');
    currentTab = tabKey;
    if (tabKey === 'traffic') { initTrafficMonth(); renderTrafficTables(); }
    else if (tabKey === 'calendar') renderCalendar();
    else if (tabKey === 'history') renderHistory();
    else renderTab(tabKey);
    if (rowId && tabKey === 'sklad') {
        setTimeout(() => {
            const row = document.querySelector(`#tbody-sklad tr[data-id="${rowId}"]`);
            if (row) {
                row.scrollIntoView({ behavior: 'smooth', block: 'center' });
                row.classList.add('search-highlight');
                setTimeout(() => row.classList.remove('search-highlight'), 3000);
            }
        }, 300);
    }
    hideSearchResults();
}

function setTrafficDateRange(range) {
    const now = new Date();
    let from, to;
    if (range === 'week') {
        const dayOfWeek = now.getDay() || 7;
        from = new Date(now); from.setDate(now.getDate() - dayOfWeek + 1);
        to = new Date(from); to.setDate(from.getDate() + 6);
    } else if (range === 'month') {
        from = new Date(now.getFullYear(), now.getMonth(), 1);
        to = new Date(now.getFullYear(), now.getMonth() + 1, 0);
    } else if (range === 'quarter') {
        const quarter = Math.floor(now.getMonth() / 3);
        from = new Date(now.getFullYear(), quarter * 3, 1);
        to = new Date(now.getFullYear(), quarter * 3 + 3, 0);
    }
    document.getElementById('traffic-date-from').value = from.toISOString().split('T')[0];
    document.getElementById('traffic-date-to').value = to.toISOString().split('T')[0];
    renderTrafficTables();
}

function calculateForecast() {
    const monthValue = document.getElementById('traffic-month')?.value || new Date().toISOString().slice(0, 7);
    const [year, month] = monthValue.split('-').map(Number);
    const daysInMonth = getDaysInMonth(month - 1, year);
    const today = new Date();
    const isCurrentMonth = today.getFullYear() === year && today.getMonth() === month - 1;
    const daysPassed = isCurrentMonth ? today.getDate() : daysInMonth;
    const skladData = tabsState.sklad?.data || [];
    let contractsSoFar = 0;
    skladData.forEach(r => {
        if (r.status !== 'Продан' || !r.dateDkp) return;
        const d = new Date(r.dateDkp);
        if (d.getFullYear() === year && d.getMonth() === month - 1 && d.getDate() <= daysPassed) contractsSoFar++;
    });
    const dailyAverage = daysPassed > 0 ? contractsSoFar / daysPassed : 0;
    const forecastTotal = Math.round(dailyAverage * daysInMonth);
    let planTotal = 0;
    for (let day = 1; day <= daysInMonth; day++) {
        const dayStr = String(day).padStart(2, '0');
        planTotal += todayPlans[monthValue]?.[dayStr]?.contracts || 0;
    }
    const prevMonth = month === 1 ? 12 : month - 1;
    const prevYear = month === 1 ? year - 1 : year;
    let prevMonthContracts = 0;
    skladData.forEach(r => {
        if (r.status !== 'Продан' || !r.dateDkp) return;
        const d = new Date(r.dateDkp);
        if (d.getFullYear() === prevYear && d.getMonth() === prevMonth - 1) prevMonthContracts++;
    });
    const trendPct = prevMonthContracts > 0 ? ((contractsSoFar / prevMonthContracts - 1) * 100).toFixed(1) : 0;
    return { contractsSoFar, daysPassed, daysInMonth, dailyAverage: dailyAverage.toFixed(1), forecastTotal, planTotal, trendPct, isCurrentMonth };
}

function renderForecast() {
    const container = document.getElementById('forecast-section');
    const forecast = calculateForecast();
    if (!forecast.isCurrentMonth) {
        container.innerHTML = `<h3>📈 Статистика месяца</h3><div class="forecast-grid"><div class="forecast-item"><div class="label">Всего контрактов</div><div class="value">${forecast.contractsSoFar}</div><div class="sub">за ${forecast.daysInMonth} дней</div></div><div class="forecast-item"><div class="label">Среднее в день</div><div class="value">${forecast.dailyAverage}</div></div></div>`;
        return;
    }
    const daysLeft = forecast.daysInMonth - forecast.daysPassed;
    const planProgress = forecast.planTotal > 0 ? ((forecast.contractsSoFar / forecast.planTotal) * 100).toFixed(0) : 0;
    container.innerHTML = `<h3>🔮 Прогноз на ${forecast.daysInMonth} дней</h3><div class="forecast-grid"><div class="forecast-item"><div class="label">Сделано</div><div class="value">${forecast.contractsSoFar}</div><div class="sub">за ${forecast.daysPassed} дн.</div></div><div class="forecast-item"><div class="label">Прогноз</div><div class="value">${forecast.forecastTotal}</div><div class="sub">~${forecast.dailyAverage}/день</div></div><div class="forecast-item"><div class="label">План</div><div class="value">${forecast.planTotal}</div><div class="sub">${planProgress}% выполнено</div></div><div class="forecast-item"><div class="label">Тренд</div><div class="value">${forecast.trendPct > 0 ? '+' : ''}${forecast.trendPct}%</div><div class="sub">vs прошлый месяц</div></div><div class="forecast-item"><div class="label">Осталось</div><div class="value">${daysLeft}</div><div class="sub">дней</div></div></div>`;
}

function renderStepForm() {
    const indicator = document.getElementById('stepIndicator');
    const content = document.getElementById('stepFormContent');
    const prevBtn = document.getElementById('stepPrevBtn');
    const nextBtn = document.getElementById('stepNextBtn');
    const totalSteps = stepFields.length;
    let dotsHtml = '';
    for (let i = 0; i < totalSteps; i++) {
        let cls = 'step-dot';
        if (i === stepForm.currentStep) cls += ' active';
        else if (i < stepForm.currentStep) cls += ' done';
        dotsHtml += `<div class="${cls}"></div>`;
    }
    indicator.innerHTML = dotsHtml;
    const step = stepFields[stepForm.currentStep];
    const val = stepForm.data[step.key] !== undefined ? stepForm.data[step.key] : (step.type === 'select' ? selectOptions[step.options]?.[0] || '' : (step.type === 'number' ? 0 : ''));
    let inputHtml = '';
    if (step.type === 'select') {
        const opts = (selectOptions[step.options] || []).map(o => `<option value="${o}" ${o === val ? 'selected' : ''}>${o}</option>`).join('');
        inputHtml = `<select id="stepInput" style="width:100%; padding:8px;">${opts}</select>`;
    } else if (step.type === 'date') inputHtml = `<input type="date" id="stepInput" value="${val}" style="width:100%; padding:8px;">`;
    else if (step.type === 'number') inputHtml = `<input type="number" id="stepInput" value="${val}" style="width:100%; padding:8px;">`;
    else if (step.type === 'textarea') inputHtml = `<textarea id="stepInput" style="width:100%; padding:8px; min-height:80px;">${val}</textarea>`;
    else inputHtml = `<input type="text" id="stepInput" value="${val}" style="width:100%; padding:8px;">`;
    content.innerHTML = `<div class="step-form"><div class="step-title">Шаг ${stepForm.currentStep + 1} из ${totalSteps}: ${step.label}</div><div>${inputHtml}</div></div>`;

    prevBtn.style.display = stepForm.currentStep > 0 ? 'inline-block' : 'none';
    nextBtn.textContent = stepForm.currentStep === totalSteps - 1 ? '✅ Сохранить' : 'Далее →';

function getDaysInMonth(month, year) { return new Date(year, month + 1, 0).getDate(); }

function getDayName(year, month, day) { return ['Вс', 'Пн', 'Вт', 'Ср', 'Чт', 'Пт', 'Сб'][new Date(year, month - 1, day).getDay()]; }

function getWeeksOfMonth(year, month) {
    const daysInMonth = getDaysInMonth(month - 1, year);
    const weeks = [];
    let currentWeek = { days: [] };
    for (let day = 1; day <= daysInMonth; day++) {
        const dayOfWeek = new Date(year, month - 1, day).getDay();
        const isMonday = dayOfWeek === 1 || day === 1;
        if (isMonday && currentWeek.days.length > 0) { weeks.push(currentWeek); currentWeek = { days: [] }; }
        currentWeek.days.push(day);
    }
    if (currentWeek.days.length > 0) weeks.push(currentWeek);
    return weeks;
}

function ensureModelData(monthValue, model) {
    if (!trafficData[monthValue]) trafficData[monthValue] = {};
    if (!trafficData[monthValue][model]) trafficData[monthValue][model] = { callsAndApps: {}, visits: {} };
    return trafficData[monthValue][model];
}

function getSkladStats(monthValue) {
    const skladData = tabsState.sklad?.data || [];
    const calls = { byModel: {}, total: 0 };
    const callsStrike = { byModel: {}, total: 0 };
    const visits = { byModel: {}, total: 0 };
    const visitsStrike = { byModel: {}, total: 0 };
    skladData.forEach(r => {
        if (!r.dateDkp || !r.dateDkp.startsWith(monthValue)) return;
        const model = r.model || 'Не указано';
        const traffic = r.traffic || '';
        const status = r.status;
        const isCallOrApp = traffic.includes('Звонок') || traffic.includes('Заявка');
        const isVisit = traffic.includes('Визит');
        if (status === 'Продан') {
            if (isCallOrApp) { if (!calls.byModel[model]) calls.byModel[model] = 0; calls.byModel[model]++; calls.total++; }
            else if (isVisit) { if (!visits.byModel[model]) visits.byModel[model] = 0; visits.byModel[model]++; visits.total++; }
        } else if (status === 'Отказ') {
            if (isCallOrApp) { if (!callsStrike.byModel[model]) callsStrike.byModel[model] = 0; callsStrike.byModel[model]++; callsStrike.total++; }
            else if (isVisit) { if (!visitsStrike.byModel[model]) visitsStrike.byModel[model] = 0; visitsStrike.byModel[model]++; visitsStrike.total++; }
        }
    });
    return { calls, callsStrike, visits, visitsStrike };
}

function getContractsByDay(monthValue, trafficFilter) {
    const skladData = tabsState.sklad?.data || [];
    const [year, month] = monthValue.split('-').map(Number);
    const daysInMonth = getDaysInMonth(month - 1, year);
    const result = {};
    for (let day = 1; day <= daysInMonth; day++) {
        const dayStr = String(day).padStart(2, '0');
        const dateStr = `${monthValue}-${dayStr}`;
        result[dayStr] = 0;
        skladData.forEach(r => {
            if (r.status !== 'Продан' || !r.dateDkp) return;
            if (r.dateDkp === dateStr) {
                const traffic = r.traffic || '';
                if (trafficFilter === 'all') result[dayStr]++;
                else if (trafficFilter === 'calls' && (traffic.includes('Звонок') || traffic.includes('Заявка'))) result[dayStr]++;
                else if (trafficFilter === 'visits' && traffic.includes('Визит')) result[dayStr]++;
            }
        });
    }
    return result;
}

function getSkladWeekContracts(year, month, days) {
    const skladData = tabsState.sklad?.data || [];
    let count = 0;
    skladData.forEach(r => {
        if (r.status !== 'Продан' || !r.dateDkp) return;
        const d = new Date(r.dateDkp);
        if (d.getFullYear() === year && d.getMonth() === month - 1 && days.includes(d.getDate())) {
            const traffic = r.traffic || '';
            if (traffic.includes('Звонок') || traffic.includes('Заявка')) count++;
        }
    });
    return count;
}

function getSkladWeekContractsVisit(year, month, days) {
    const skladData = tabsState.sklad?.data || [];
    let count = 0;
    skladData.forEach(r => {
        if (r.status !== 'Продан' || !r.dateDkp) return;
        const d = new Date(r.dateDkp);
        if (d.getFullYear() === year && d.getMonth() === month - 1 && days.includes(d.getDate())) {
            if ((r.traffic || '').includes('Визит')) count++;
        }
    });
    return count;
}

function exportTrafficCSV() {
    const monthValue = document.getElementById('traffic-month').value;
    if (!monthValue) { showToast('Выберите месяц', 'error'); return; }
    const [year, month] = monthValue.split('-').map(Number);
    const daysInMonth = getDaysInMonth(month - 1, year);
    const models = selectOptions.model || [];
    const monthData = trafficData[monthValue] || {};
    const stats = getSkladStats(monthValue);
    let csv = 'ТРАФИК - ' + monthValue + '\n\nЗВОНКИ\nМодель';
    for (let day = 1; day <= daysInMonth; day++) csv += `;День ${day}`;
    csv += ';Итого;Контракты;%\n';
    models.forEach(model => {
        const md = monthData[model] || { callsAndApps: {}, visits: {} };
        csv += model;
        let total = 0;
        for (let day = 1; day <= daysInMonth; day++) {
            const v = md.callsAndApps[String(day).padStart(2, '0')] || 0;
            csv += `;${v}`;
            total += v;
        }
        const pct = total > 0 ? ((stats.calls.byModel[model] || 0) / total * 100).toFixed(1) + '%' : '—';
        csv += `;${total};${stats.calls.byModel[model] || 0};${pct}\n`;
    });
    const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `traffic_${monthValue}.csv`;
    a.click();
    URL.revokeObjectURL(url);
    showToast('📥 CSV скачан');
}

function exportPlanFactToExcel() {
    const monthKey = `${currentPlanFactYear}-${String(currentPlanFactMonth + 1).padStart(2, '0')}`;
    const daysInMonth = getDaysInMonth(currentPlanFactMonth, currentPlanFactYear);
    const excelData = [];
    const headers = ['Канал', 'Группа', 'Бюджет', 'CPL', 'РЛ', 'SR%'];
    for (let day = 1; day <= daysInMonth; day++) headers.push(`День ${day}`);
    headers.push('Итого лидов', 'Контракты (план)', 'Выдачи (план)', 'Контракты (факт)', 'Выдачи (факт)');
    excelData.push(headers);
    channelsData.forEach(channel => {
        const plan = planData[monthKey][channel.name];
        const row = [channel.name, channel.group, plan.budget, plan.cpl, plan.rl, plan.sr];
        let totalLeads = 0;
        for (let day = 1; day <= daysInMonth; day++) {
            const value = plan.days[day] || 0;
            row.push(value);
            totalLeads += value;
        }
        row.push(totalLeads);
        row.push(Math.round(plan.rl * plan.sr / 100));
        row.push(Math.round(totalLeads * plan.sr / 100));
        const fact = factData[monthKey];
        row.push(fact.contracts || 0);
        row.push(fact.issued || 0);
        excelData.push(row);
    });
    const wb = XLSX.utils.book_new();
    const ws = XLSX.utils.aoa_to_sheet(excelData);
    XLSX.utils.book_append_sheet(wb, ws, `${MONTHS[currentPlanFactMonth]} ${currentPlanFactYear}`);
    XLSX.writeFile(wb, `plan_fact_${monthKey}.xlsx`);
    showToast('📊 Excel экспортирован');
}

document.addEventListener('keydown', (e) => {
    if (e.key === '?' && !['INPUT', 'TEXTAREA', 'SELECT'].includes(document.activeElement.tagName)) {
        document.getElementById('shortcutsHint').classList.toggle('active');
        return;
    }
    if (e.ctrlKey && e.key === 'a' && currentTab === 'sklad') {
        if (['INPUT', 'TEXTAREA'].includes(document.activeElement.tagName)) return;
        e.preventDefault();
        const checkbox = document.querySelector('#thead-sklad .checkbox-cell input');
        if (checkbox) { checkbox.checked = !checkbox.checked; toggleAllRows('sklad', checkbox); }
        return;
    }
    if (e.ctrlKey && e.key === 's') { e.preventDefault(); saveFullBackup(); return; }
    if (e.ctrlKey && e.key === 'f') { e.preventDefault(); document.getElementById('globalSearchInput').focus(); return; }
    if (e.ctrlKey && e.key === 'p') { e.preventDefault(); printReport(); return; }
    if (!['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight'].includes(e.key)) return;
    const activeEl = document.activeElement;
    if (!activeEl) return;
    const isTrafficCell = activeEl.matches('td[contenteditable="true"]') || activeEl.matches('td.plan-cell input') || activeEl.matches('td.day-cell');
    if (!isTrafficCell) return;
    e.preventDefault();
    const currentCell = activeEl.closest('td');
    if (!currentCell) return;
    const currentRow = currentCell.closest('tr');
    if (!currentRow) return;
    const table = currentCell.closest('table');
    if (!table) return;
    const allRows = Array.from(table.querySelectorAll('tr'));
    const currentRowIndex = allRows.indexOf(currentRow);
    const allCells = Array.from(currentRow.querySelectorAll('td'));
    const currentCellIndex = allCells.indexOf(currentCell);
    let targetRow = currentRowIndex;
    let targetCell = currentCellIndex;
    if (e.key === 'ArrowUp') targetRow = Math.max(0, currentRowIndex - 1);
    else if (e.key === 'ArrowDown') targetRow = Math.min(allRows.length - 1, currentRowIndex + 1);
    else if (e.key === 'ArrowLeft') targetCell = Math.max(0, currentCellIndex - 1);
    else if (e.key === 'ArrowRight') targetCell = Math.min(allCells.length - 1, currentCellIndex + 1);
    const targetRowEl = allRows[targetRow];
    if (!targetRowEl) return;
    const targetCells = Array.from(targetRowEl.querySelectorAll('td'));
    const targetCellEl = targetCells[targetCell];
    if (!targetCellEl) return;
    const editableEl = targetCellEl.querySelector('[contenteditable="true"]') || targetCellEl.querySelector('input') || targetCellEl;
    if (editableEl) {
        editableEl.focus();
        if (editableEl.tagName === 'INPUT') editableEl.select();
    }
});

    const cfg = TABS_CONFIG[tabKey];

    if (!cfg || !state) return;

function renderFilters(tabKey) {
    const state = tabsState[tabKey];
    const filters = document.getElementById(`filters-${tabKey}`);
    const selectCols = state.columns.filter(c => c.type === 'select');
    let html = '';
    selectCols.slice(0, 4).forEach(c => {
        const values = [...new Set(state.data.map(r => r[c.key]).filter(Boolean))].sort();
        html += `<label>${c.label}: <select id="filter-${tabKey}-${c.key}" onchange="renderBody('${tabKey}'); updateStats('${tabKey}')"><option value="">Все</option>${values.map(v => `<option>${v}</option>`).join('')}</select></label>`;
    });
    filters.innerHTML = html;
}

    const state = tabsState[tabKey];

function renderHeader(tabKey) {
    const thead = document.getElementById(`thead-${tabKey}`);
    const state = tabsState[tabKey];
    let html = '<th class="row-num">№</th><th class="checkbox-cell"><input type="checkbox" onchange="toggleAllRows(\'' + tabKey + '\', this)"></th>';
    state.columns.forEach((c, i) => {
        html += `<th data-key="${c.key}" title="${c.label}"><span class="th-label" ondblclick="startRenameColumn(event, '${tabKey}', '${c.key}')" oncontextmenu="showHeaderContextMenu(event, '${tabKey}', '${c.key}')">${c.label}</span><div class="col-resizer" data-col-index="${i}" onmousedown="startResize(event, '${tabKey}', ${i})"></div></th>`;

}

function getFilteredData(tabKey) {
    const state = tabsState[tabKey];
    const selectCols = state.columns.filter(c => c.type === 'select').slice(0, 4);
    return state.data.filter(r => {
        for (const c of selectCols) {
            const selectEl = document.getElementById(`filter-${tabKey}-${c.key}`);
            if (selectEl && selectEl.value && r[c.key] !== selectEl.value) return false;
        }
        return true;
    });
}

function renderBody(tabKey) {
    const tbody = document.getElementById(`tbody-${tabKey}`);
    const cfg = TABS_CONFIG[tabKey];
    const state = tabsState[tabKey];
    const filtered = getFilteredData(tabKey);
    let html = '';
    filtered.forEach((r, i) => {
        let rowClass = '';
        if (cfg.statusField && r[cfg.statusField] && cfg.statusMap[r[cfg.statusField]] !== undefined) rowClass = cfg.statusMap[r[cfg.statusField]];
        const isSelected = selectedRows.has(`${tabKey}_${r.id}`);
        html += `<tr class="${rowClass}" data-id="${r.id}" onclick="selectRow('${tabKey}', ${r.id})" oncontextmenu="showRowContextMenu(event, '${tabKey}', ${r.id})">`;
        html += `<td class="row-num">${i + 1}</td>`;
        html += `<td class="checkbox-cell" onclick="event.stopPropagation()"><input type="checkbox" ${isSelected ? 'checked' : ''} onchange="toggleRowSelection('${tabKey}', ${r.id}, this)"></td>`;
        state.columns.forEach(c => html += renderCell(tabKey, r, c));
        html += `</tr>`;
    });
    html += renderTotalRow(tabKey, filtered);
    tbody.innerHTML = html;
}

function renderCell(tabKey, r, c) {
    const val = r[c.key];
    if (c.type === 'select') {
        const options = (selectOptions[c.options] || []).map(o => `<option ${String(o) === String(val) ? 'selected' : ''}>${o}</option>`).join('');
        let extraClass = '';
        if (c.key === 'ti' && tabKey === 'sklad') {
            const hasLink = !!evaluationLinks[r.id];
            const tiClass = hasLink ? 'cell-ti has-link' : 'cell-ti';
            return `<td class="${tiClass}" oncontextmenu="showTiContextMenu(event, ${r.id})"><select onchange="updateCell('${tabKey}', ${r.id}, '${c.key}', this.value)">${options}</select></td>`;
        }
        return `<td class="${extraClass}"><select onchange="updateCell('${tabKey}', ${r.id}, '${c.key}', this.value)">${options}</select></td>`;
    }
    if (c.type === 'number') {
        const cls = val < 0 ? 'negative' : (val > 0 ? 'positive' : '');
        const display = val ? new Intl.NumberFormat('ru-RU').format(val) : '0';
        return `<td class="cell-number ${cls}" contenteditable="true" onblur="updateCellNum('${tabKey}', ${r.id}, '${c.key}', this.textContent)">${display}</td>`;
    }
    if (c.type === 'date') return `<td><input type="date" value="${val || ''}" onchange="updateCell('${tabKey}', ${r.id}, '${c.key}', this.value)"></td>`;
    return `<td class="cell-text" contenteditable="true" onblur="updateCell('${tabKey}', ${r.id}, '${c.key}', this.textContent)">${val || ''}</td>`;
}

function renderTotalRow(tabKey, rows) {
    const state = tabsState[tabKey];
    let totals = {};
    state.columns.forEach(c => { if (c.type === 'number') totals[c.key] = 0; });
    rows.forEach(r => { state.columns.forEach(c => { if (c.type === 'number') totals[c.key] += r[c.key] || 0; }); });
    let html = `<tr class="total-row"><td class="row-num">Σ</td><td></td>`;
    state.columns.forEach(c => {
        if (c.type === 'number') {
            const cls = totals[c.key] < 0 ? 'negative' : 'positive';
            html += `<td class="cell-number ${cls}">${new Intl.NumberFormat('ru-RU').format(totals[c.key])}</td>`;
        } else html += `<td></td>`;
    });
    html += `</tr>`;
    return html;
}

function updateCell(tabKey, id, key, value) {
    const r = tabsState[tabKey].data.find(r => r.id === id);
    if (!r) return;
    const oldValue = r[key];
    r[key] = value.trim ? value.trim() : value;
    saveTabDataKey(tabKey);
    renderBody(tabKey);
    updateStats(tabKey);
    if (oldValue !== r[key]) addHistoryEntry('edit', `${TABS_CONFIG[tabKey].name}: ${r.model || ''} - ${key} изменено`);
}

function updateCellNum(tabKey, id, key, text) {
    const r = tabsState[tabKey].data.find(r => r.id === id);
    if (!r) return;
    const oldValue = r[key];
    r[key] = parseFloat(text.replace(/\s/g, '').replace(',', '.')) || 0;
    saveTabDataKey(tabKey);
    renderBody(tabKey);
    updateStats(tabKey);
    if (oldValue !== r[key]) addHistoryEntry('edit', `${TABS_CONFIG[tabKey].name}: ${r.model || ''} - ${key} = ${r[key]}`);
}

function duplicateRow(tabKey) {
    if (!contextRowId && !selectedRowId) { showToast('Выберите строку', 'error'); return; }
    const id = contextRowId || selectedRowId;
    const state = tabsState[tabKey];
    const r = state.data.find(r => r.id === id);
    if (!r) return;
    const newRow = { ...r, id: Date.now() };
    const idx = state.data.findIndex(r => r.id === id);
    state.data.splice(idx + 1, 0, newRow);
    if (evaluationLinks[id]) { evaluationLinks[newRow.id] = evaluationLinks[id]; saveEvaluationLinks(); }
    saveTabDataKey(tabKey);
    renderBody(tabKey);
    updateStats(tabKey);
    showToast('📋 Дублировано');
    addHistoryEntry('add', `${TABS_CONFIG[tabKey].name}: дублирована строка`);
}

    if (!contextRowId) return;
    const state = tabsState[tabKey];

function updateStats(tabKey) {
    const state = tabsState[tabKey];
    const cfg = TABS_CONFIG[tabKey];
    const statsEl = document.getElementById(`stats-${tabKey}`);
    if (!statsEl) return;
    const filtered = getFilteredData(tabKey);
    const total = filtered.length;
    let statusHtml = '';
    if (cfg.statusField) {
        const counts = {};
        filtered.forEach(r => { const v = r[cfg.statusField]; if (v) counts[v] = (counts[v] || 0) + 1; });
        statusHtml = Object.entries(counts).map(([k, v]) => `${k}: <b>${v}</b>`).join(' | ');
    }
    statsEl.innerHTML = `Всего: <b>${total}</b> ${statusHtml ? '| ' + statusHtml : ''}`;
}

    const state = tabsState[tabKey];
    const c = state.columns[colIndex];
    if (!c) return;
    resizer.classList.add('active');
    document.body.classList.add('resizing');
    resizeState = { tabKey, colIndex, startX: e.clientX, startWidth: c.width, resizer };
    document.addEventListener('mousemove', doResize);
    document.addEventListener('mouseup', stopResize);
}

function doResize(e) {
    if (!resizeState) return;
    const diff = e.clientX - resizeState.startX;
    const newWidth = Math.max(MIN_COL_WIDTH, resizeState.startWidth + diff);
    tabsState[resizeState.tabKey].columns[resizeState.colIndex].width = newWidth;
    const colgroup = document.getElementById(`colgroup-${resizeState.tabKey}`);
    const cols = colgroup.querySelectorAll('col');
    if (cols[resizeState.colIndex + 2]) cols[resizeState.colIndex + 2].style.width = newWidth + 'px';
}

function stopResize() {
    if (!resizeState) return;
    resizeState.resizer.classList.remove('active');
    document.body.classList.remove('resizing');
    document.removeEventListener('mousemove', doResize);
    document.removeEventListener('mouseup', stopResize);
    saveTabColumns(resizeState.tabKey);
    resizeState = null;
}

function exportCSV(tabKey) {
    const state = tabsState[tabKey];
    const headers = ['№', ...state.columns.map(c => c.label), 'Ссылка'];
    let csv = headers.join(';') + '\n';
    state.data.forEach((r, i) => {
        const row = [i + 1, ...state.columns.map(c => r[c.key] || ''), evaluationLinks[r.id] || ''];
        csv += row.map(v => `"${(v || '').toString().replace(/"/g, '""')}"`).join(';') + '\n';
    });
    const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `${tabKey}_${new Date().toISOString().split('T')[0]}.csv`;
    a.click();
    URL.revokeObjectURL(url);
    showToast('📥 CSV скачан');
}

function importCSV(event, tabKey) {
    const file = event.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = e => {
        try {
            const lines = e.target.result.split('\n').slice(1);
            const imported = [];
            lines.forEach((line, i) => {
                if (!line.trim()) return;
                const parts = line.match(/("([^"]*("")*)*"|[^;]+)/g) || [];
                const clean = parts.map(p => p.replace(/^"|"$/g, '').replace(/""/g, '"'));
                if (clean.length < 2) return;
                const r = { id: Date.now() + i };
                tabsState[tabKey].columns.forEach((c, idx) => {
                    const v = clean[idx + 1] || '';
                    r[c.key] = c.type === 'number' ? (parseFloat(v.replace(/\s/g, '').replace(',', '.')) || 0) : v;
                });
                imported.push(r);
                const linkValue = clean[tabsState[tabKey].columns.length + 1];
                if (linkValue && linkValue.trim()) evaluationLinks[r.id] = linkValue.trim();
            });
            if (imported.length > 0) {
                tabsState[tabKey].data = imported;
                saveTabDataKey(tabKey);
                saveEvaluationLinks();
                renderTab(tabKey);
                showToast(`📤 Импортировано ${imported.length}`);
                addHistoryEntry('bulk', `Импорт ${imported.length} строк из CSV`);
            }
        } catch (err) { showToast('❌ ' + err.message, 'error'); }
    };
    reader.readAsText(file, 'UTF-8');
    event.target.value = '';
}

function applyAnalyticsFilters() {
    const data = tabsState.sklad?.data || [];
    const statusFilter = document.getElementById('filter-status')?.value || 'all';
    const modelFilter = document.getElementById('filter-model')?.value || 'all';
    const trafficFilter = document.getElementById('filter-traffic')?.value || 'all';
    const riskFilter = document.getElementById('filter-risk')?.value || 'all';
    const krFilter = document.getElementById('filter-kr')?.value || 'all';
    const tiFilter = document.getElementById('filter-ti')?.value || 'all';
    const filtered = data.filter(r => {
        if (statusFilter !== 'all' && r.status !== statusFilter) return false;
        if (modelFilter !== 'all' && r.model !== modelFilter) return false;

    skladData.forEach(r => {
        if (r.dateDkp && r.dateDkp.startsWith(prefix) && r.status === 'Продан') {
            if (!factByDate[r.dateDkp]) factByDate[r.dateDkp] = { contracts: 0, issued: 0 };
            factByDate[r.dateDkp].contracts++;
        }
        if (r.dateIssued && r.dateIssued.startsWith(prefix) && r.status === 'Продан') {
            if (!factByDate[r.dateIssued]) factByDate[r.dateIssued] = { contracts: 0, issued: 0 };
            factByDate[r.dateIssued].issued++;
        }
    });
    const firstDay = new Date(year, month, 1);
    const lastDay = new Date(year, month + 1, 0);
    const startDayOfWeek = (firstDay.getDay() + 6) % 7;
    const daysInMonth = lastDay.getDate();
    const today = new Date();
    const todayStr = today.toISOString().split('T')[0];
    let html = `<div class="calendar-controls"><h3>📅 ${monthName.charAt(0).toUpperCase() + monthName.slice(1)}</h3><div class="calendar-nav-btns"><button onclick="changeCalendarMonth(-1)">◀ Пред.</button><button onclick="changeCalendarMonth(0)">Сегодня</button><button onclick="changeCalendarMonth(1)">След. ▶</button></div></div><div class="calendar-grid"><div class="calendar-day-header">Пн</div><div class="calendar-day-header">Вт</div><div class="calendar-day-header">Ср</div><div class="calendar-day-header">Чт</div><div class="calendar-day-header">Пт</div><div class="calendar-day-header">Сб</div><div class="calendar-day-header">Вс</div>`;
    for (let i = 0; i < startDayOfWeek; i++) html += '<div class="calendar-day other-month"></div>';
    for (let day = 1; day <= daysInMonth; day++) {
        const dateStr = `${prefix}-${String(day).padStart(2, '0')}`;
        const isToday = dateStr === todayStr;
        const tp = todayPlans[prefix]?.[String(day).padStart(2, '0')] || { meetings: 0, contracts: 0 };
        const fact = factByDate[dateStr] || { contracts: 0, issued: 0 };
        html += `<div class="calendar-day ${isToday ? 'today' : ''}" onclick="openCalendarDay('${dateStr}')"><div class="day-number">${day}</div><div class="day-stats"><div class="calendar-stat plan-meetings"><span class="stat-label">📅 Встречи:</span><span class="stat-value">${tp.meetings || 0}</span></div><div class="calendar-stat plan-contracts"><span class="stat-label">📋 План:</span><span class="stat-value">${tp.contracts || 0}</span></div><div class="calendar-stat fact-contracts"><span class="stat-label">✅ Факт:</span><span class="stat-value">${fact.contracts}</span></div><div class="calendar-stat fact-issued"><span class="stat-label">🚗 Выдачи:</span><span class="stat-value">${fact.issued}</span></div></div></div>`;
    }
    html += '</div>';
    content.innerHTML = html;
}

function changeCalendarMonth(delta) {
    if (delta === 0) calendarCurrentDate = new Date();
    else calendarCurrentDate.setMonth(calendarCurrentDate.getMonth() + delta);
    renderCalendar();
}

function openCalendarDay(dateStr) {
    const [year, month, day] = dateStr.split('-');
    const monthKey = `${year}-${month}`;
    const tp = todayPlans[monthKey]?.[day] || { meetings: 0, contracts: 0 };
    const fact = getFactForDate(dateStr);
    document.getElementById('calendarDayTitle').textContent = `📅 ${new Date(dateStr).toLocaleDateString('ru-RU', { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' })}`;
    document.getElementById('calendarDayContent').innerHTML = `<div class="day-detail"><label>📞 Встречи:</label><input type="number" value="${tp.meetings || 0}" onchange="updateTodayPlanDate('${monthKey}', '${day}', 'meetings', this.value)"></div><div class="day-detail"><label>📋 План контрактов:</label><input type="number" value="${tp.contracts || 0}" onchange="updateTodayPlanDate('${monthKey}', '${day}', 'contracts', this.value)"></div><div style="margin-top:15px; padding:10px; background:#f8f9fa; border-radius:6px;"><div style="font-size:12px; font-weight:600; margin-bottom:8px;">📊 Факт за ${dateStr}:</div><div style="font-size:11px; margin-bottom:4px;">✅ Контрактов: <b>${fact.contracts}</b></div><div style="font-size:11px;">🚗 Выдач: <b>${fact.issued}</b></div></div>`;
    document.getElementById('calendarDayModal').classList.add('active');
}

function closeCalendarDayModal() { document.getElementById('calendarDayModal').classList.remove('active'); renderCalendar(); }

function updateTodayPlanDate(monthKey, day, key, value) {
    if (!todayPlans[monthKey]) todayPlans[monthKey] = {};
    if (!todayPlans[monthKey][day]) todayPlans[monthKey][day] = { meetings: 0, contracts: 0 };
    todayPlans[monthKey][day][key] = parseInt(value) || 0;
    saveTodayPlans();
}

function getFactForDate(dateStr) {
    const skladData = tabsState.sklad?.data || [];
    let contracts = 0, issued = 0;
    skladData.forEach(r => {
        if (r.status === 'Продан') {
            if (r.dateDkp === dateStr) contracts++;
            if (r.dateIssued === dateStr) issued++;
        }
    });
    return { contracts, issued };
}

function showToast(msg, type = 'success') {
    const t = document.createElement('div');
    t.className = 'toast' + (type === 'error' ? ' error' : type === 'warning' ? ' warning' : '');
    t.textContent = msg;
    document.body.appendChild(t);
    setTimeout(() => t.remove(), 3000);
}

function updateCurrentDate() {
    const now = new Date();
    document.getElementById('currentDate').textContent = now.toLocaleDateString('ru-RU', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
}
